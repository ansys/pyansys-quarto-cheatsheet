% Optimized code block styling for cheat sheet density
\lstdefinestyle{python_style}{
    backgroundcolor=\color{shadecolor}, % Set the background color
    commentstyle=\color{codegreen}, % Style for comments
    keywordstyle=\color{magenta}, % Style for keywords
    numberstyle=\tiny\color{codegray}, % Style for line numbers
    stringstyle=\color{codepurple}, % Style for strings
    breaklines=true, % Enable line breaking
    breakatwhitespace=false, % Allow long tokens (for example imports) to wrap
    columns=fullflexible, % Improve wrapping/alignment in narrow columns
    keepspaces=true, % Preserve spacing while still allowing line wraps
    breakindent=0pt, % Avoid extra indentation that can push text into next column
    linewidth=\linewidth, % Constrain listings to current column width
    resetmargins=true, % Ensure no inherited margins widen the listing box
    basicstyle=\ttfamily\scriptsize, % Smaller font for more content
    frame=single, % Add a frame around the code
    framesep=1pt, % padding between frame and code
    xleftmargin=1pt, % No left margin for maximum width
    rulecolor=\color{shadecolor},
    showstringspaces=false, % Don't show spaces as special characters
    tabsize=1, % Smaller tab size
}
\lstset{
    style=python_style,
    upquote=true,
}

}
